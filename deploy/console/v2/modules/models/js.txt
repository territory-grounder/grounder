/* namespace-aware SVG helper: use the chart-primitives global if present */
const mdlSvg = (typeof svgEl === "function") ? svgEl :
  function (tag, attrs = {}, ...kids) {
    const e = document.createElementNS("http://www.w3.org/2000/svg", tag);
    for (const [k, v] of Object.entries(attrs)) {
      if (v == null || v === false) continue;
      e.setAttribute(k, v === true ? "" : String(v));
    }
    for (const k of kids.flat()) { if (k == null || k === false) continue; e.append(k.nodeType ? k : document.createTextNode(k)); }
    return e;
  };

const mdlFmtTok = n => n >= 1e6 ? (n / 1e6).toFixed(n >= 1e8 ? 0 : 1).replace(/\.0$/, "") + "M"
  : n >= 1e3 ? (n / 1e3).toFixed(1).replace(/\.0$/, "") + "k" : String(n);
const mdlFmtInt = n => n.toLocaleString("en-US");
const mdlUsd = n => "$" + n.toFixed(2);

/* ---------------- small pieces ---------------- */
function mdlSec(lbl, title, ...extra) {
  return h("div", { class: "mdl-sec" }, h("span", { class: "lbl" }, lbl), h("h3", {}, title),
    h("span", { class: "mdl-rule" }), ...extra);
}
function mdlProv(p) { return h("span", { class: "mdl-prov" }, p); }
function mdlChain(primary, fallbacks) {
  const c = h("span", { class: "mdl-chain" });
  c.append(h("span", { class: "mdl-model pri", title: "primary" }, primary));
  (fallbacks || []).forEach(f => c.append(
    h("span", { class: "mdl-arr", "aria-hidden": "true" }, "→"),
    h("span", { class: "mdl-model fb", title: "fallback" }, f)));
  return c;
}
function mdlCard(capKids, bodyNode) {
  const card = h("div", { class: "mdl-card" });
  card.append(h("div", { class: "cap" }, ...capKids));
  if (bodyNode) card.append(bodyNode);
  return card;
}
function mdlTile(lbl, big, sub, cls) {
  return h("div", { class: "mdl-tile" },
    h("div", { class: "lbl" }, lbl),
    h("div", { class: "mdl-big" + (cls ? " " + cls : "") }, big),
    sub ? h("div", { class: "mdl-sub" }, sub) : null);
}

/* ---------------- charts (built on svgEl; tokens only) ---------------- */
function mdlArcPath(cx, cy, r, a0, a1) {
  const large = (a1 - a0) > Math.PI ? 1 : 0;
  return "M " + (cx + r * Math.cos(a0)).toFixed(2) + " " + (cy + r * Math.sin(a0)).toFixed(2) +
    " A " + r + " " + r + " 0 " + large + " 1 " +
    (cx + r * Math.cos(a1)).toFixed(2) + " " + (cy + r * Math.sin(a1)).toFixed(2);
}
function mdlDonut(items, total) {
  const cx = 75, cy = 75, r = 57, w = 15, gap = 0.05;
  const svg = mdlSvg("svg", { viewBox: "0 0 150 150", class: "mdl-donut-svg", role: "img", "aria-label": "estimated cost by model" });
  let a = -Math.PI / 2;
  items.forEach(it => {
    const a1 = a + (it.cost / total) * 2 * Math.PI;
    svg.append(mdlSvg("path", {
      d: mdlArcPath(cx, cy, r, a + gap / 2, Math.max(a + gap / 2 + 0.01, a1 - gap / 2)),
      fill: "none", stroke: MDL_COLOR[it.id] || "var(--ink4)", "stroke-width": w,
    }));
    a = a1;
  });
  const fig = h("div", { class: "mdl-donut" }, svg,
    h("div", { class: "mdl-donut-hole" },
      h("div", { class: "mdl-donut-big" }, mdlUsd(total)),
      h("div", { class: "lbl", style: "font-size:8.5px" }, "est. spend")));
  const leg = h("div", { class: "mdl-leg" });
  items.forEach(it => {
    const c = MDL_COLOR[it.id] || "var(--ink4)";
    leg.append(h("div", { class: "mdl-leg-r" },
      h("i", { class: "sw", style: it.fallbackOnly       /* shape-codes the off-family fallback provider */
        ? "background:transparent;border:1.5px solid " + c + ";border-radius:50%" : "background:" + c }),
      h("span", { class: "mdl-leg-id" }, it.id + (it.fallbackOnly ? " (fb)" : "")),
      h("span", { class: "mdl-leg-v" }, mdlUsd(it.cost)),
      h("span", { class: "mdl-leg-p" }, pct(it.cost / total) + "%")));
  });
  return h("div", { class: "mdl-donut-wrap" }, fig, leg);
}

function mdlHbar(rows) {
  const max = Math.max(...rows.map(r => r.tin + r.tout));
  const el = h("div", { class: "mdl-hbar" });
  rows.forEach(r => {
    el.append(h("div", { class: "mdl-hb-row" },
      h("span", { class: "mdl-hb-lab" }, r.role),
      h("div", { class: "mdl-hb-track", role: "img", "aria-label": r.role + ": " + mdlFmtTok(r.tin) + " prompt, " + mdlFmtTok(r.tout) + " completion tokens" },
        h("i", { class: "tin", style: "width:" + (r.tin / max * 100).toFixed(2) + "%" }),
        h("i", { class: "tout", style: "width:" + (r.tout / max * 100).toFixed(2) + "%" })),
      h("span", { class: "mdl-hb-val" }, mdlFmtTok(r.tin) + " in · " + mdlFmtTok(r.tout) + " out")));
  });
  el.append(h("div", { class: "mdl-hb-leg" },
    h("span", {}, h("i", { class: "tin" }), "prompt"),
    h("span", {}, h("i", { class: "tout" }), "completion"),
    h("span", { class: "lbl", style: "margin-left:auto;font-size:8.5px" }, "observed · territory")));
  return el;
}

function mdlReqChart(S) {
  const W = 1080, H = 230, L = 46, R = 16, T = 18, B = 30;
  const n = S.obs.length;
  const rawMax = Math.max(...S.obs, ...S.envHi) * 1.12;
  const step = rawMax > 1000 ? 500 : 50;
  const maxY = Math.ceil(rawMax / step) * step;
  const X = i => L + (W - L - R) * (n === 1 ? 0 : i / (n - 1));
  const Y = v => T + (H - T - B) * (1 - v / maxY);
  const svg = mdlSvg("svg", { viewBox: "0 0 " + W + " " + H, class: "mdl-area", role: "img",
    "aria-label": "requests per " + S.bucket + " — observed (steel, solid) vs predicted envelope (indigo, dashed)" });
  for (let k = 0; k <= 4; k++) {
    const v = maxY * k / 4, gy = Y(v);
    svg.append(mdlSvg("line", { x1: L, x2: W - R, y1: gy.toFixed(1), y2: gy.toFixed(1), class: "mdl-grid" }));
    svg.append(mdlSvg("text", { x: L - 6, y: (gy + 3).toFixed(1), class: "mdl-ax", "text-anchor": "end" },
      v >= 1000 ? mdlFmtTok(v) : String(Math.round(v))));
  }
  S.xlabels.forEach(([i, t]) => svg.append(
    mdlSvg("text", { x: X(i).toFixed(1), y: H - 8, class: "mdl-ax", "text-anchor": i === 0 ? "start" : (i === n - 1 ? "end" : "middle") }, t)));
  const line = arr => arr.map((v, i) => (i ? "L " : "M ") + X(i).toFixed(1) + " " + Y(v).toFixed(1)).join(" ");
  const dHi = line(S.envHi), dLo = line(S.envLo);
  const dBand = dHi + " " + S.envLo.map((v, j, a) => { const i = a.length - 1 - j; return "L " + X(i).toFixed(1) + " " + Y(a[i]).toFixed(1); }).join(" ") + " Z";
  svg.append(mdlSvg("path", { d: dBand, class: "mdl-band" }));
  svg.append(mdlSvg("path", { d: dHi, class: "mdl-envline" }));
  svg.append(mdlSvg("path", { d: dLo, class: "mdl-envline" }));
  const dObs = line(S.obs);
  svg.append(mdlSvg("path", { d: dObs + " L " + X(n - 1).toFixed(1) + " " + Y(0).toFixed(1) + " L " + X(0).toFixed(1) + " " + Y(0).toFixed(1) + " Z", class: "mdl-obsfill" }));
  svg.append(mdlSvg("path", { d: dObs, class: "mdl-obsline" }));
  let noted = false;
  S.obs.forEach((v, i) => {
    if (v <= S.envHi[i]) return;
    svg.append(mdlSvg("path", { d: "M " + X(i).toFixed(1) + " " + (Y(v) - 14).toFixed(1) + " l 5 8 h -10 Z", class: "mdl-devmk" }));
    if (!noted) { noted = true;
      const tx = Math.min(X(i) + 10, W - R - 150);
      svg.append(mdlSvg("text", { x: tx.toFixed(1), y: (Y(v) - 8).toFixed(1), class: "mdl-devtxt" }, S.note));
    }
  });
  return h("div", {}, svg, h("div", { class: "mdl-chart-leg" },
    h("span", { class: "s" }, h("i", { class: "ln" }), "observed req/" + S.bucket + " · territory"),
    h("span", { class: "s" }, h("i", { class: "ln map" }), "predicted envelope · map"),
    h("span", { class: "s" }, h("i", { class: "tri" }), "over envelope → gateway fallback")));
}

/* ---------------- nav registration (resilient) ----------------
   Prefers the shell's addNav(); falls back to a namespace-correct manual
   insert if the shell helper is absent or throws (e.g. h("svg",{class})
   hitting the read-only SVGElement.className in strict mode). */
function mdlAddNav(group, viewKey, label, iconInner, badgeKey) {
  const rail = $("#rail");
  const already = () => rail && rail.querySelector('.navi[data-view="' + viewKey + '"]');
  if (typeof addNav === "function") {
    try { addNav(group, viewKey, label, iconInner, badgeKey); } catch (e) { /* fall through */ }
    if (already()) return;
  }
  if (!rail || already()) return;
  const foot = rail.querySelector(".foot");
  let grp = [...rail.querySelectorAll(".lbl.grp")].find(g => g.textContent.trim().toLowerCase() === group.toLowerCase());
  let anchor = foot;
  if (!grp) { grp = h("div", { class: "lbl grp" }, group); rail.insertBefore(grp, foot || null); }
  else {
    anchor = grp.nextElementSibling;
    while (anchor && anchor !== foot && !anchor.classList.contains("grp")) anchor = anchor.nextElementSibling;
    if (!anchor) anchor = foot;
  }
  const ic = mdlSvg("svg", { class: "ic", viewBox: "0 0 16 16", fill: "none", stroke: "currentColor", "stroke-width": "1.4" });
  ic.innerHTML = iconInner;
  const a = h("a", { class: "navi", "data-view": viewKey, href: "#" + viewKey,
    onclick: e => { e.preventDefault(); if (typeof route === "function") route(viewKey); } },
    ic, h("span", { class: "t" }, label),
    badgeKey ? h("span", { class: "n", "data-badge": badgeKey }, "—") : null);
  rail.insertBefore(a, anchor || null);
}
